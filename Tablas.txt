DROP SCHEMA public CASCADE;
CREATE SCHEMA public;

create table Fabricante(
id serial primary key,
nombre varchar(100) not null,
direccion varchar (255)
);

create table Marca(
id serial primary key,
nombre varchar(100) not null,
id_fabricante integer references Fabricante(id) on delete cascade 
);

create table Categoria(
id serial primary key,
nombre varchar (100) not null,
categoria_padre integer references Categoria(id) on delete cascade
);


create table Producto(
id serial primary key,
nombre varchar (200) not null,
tamanio varchar (100),
medida varchar (50),
id_marca integer references Marca(id) on delete cascade
);

create table Sucursal(
id serial primary key,
nombre varchar (255) not null,
direccion varchar (255) not null
);

create table Cliente (
id integer primary key,
nombre varchar(100) not null,
apellido varchar (100) not null,
direccion varchar (2555),
email varchar (255),
telefono varchar (20),
es_socio boolean not null
);

create table Factura (
id serial primary key,
fecha date not null,
id_sucursal integer references Sucursal(id) on delete cascade,
id_cliente integer references Cliente(id) on delete cascade
);

create table Inventario(
id serial primary key,
id_sucursal integer references Sucursal(id) on delete cascade,
id_producto integer references Producto(id) on delete cascade,
cantidad integer not null,
fecha date,
precio_base decimal (10,2) not null
);

create table Producto_Categoria(
id_producto integer references Producto(id) on delete cascade,
id_categoria integer references Categoria(id) on delete cascade,
primary key (id_producto,id_categoria)
);

create table Factura_Producto(
id_producto integer references Producto(id) on delete cascade,
id_factura integer references Factura(id) on delete cascade,
cantidad integer not null,
precio_unitario decimal(10,2) not null,
primary key(id_producto,id_factura)
);

create table InventarioFactura(
id_sucursal integer references Sucursal(id) on delete cascade,
id_producto integer references Producto(id) on delete cascade,
cantidad integer,
precio decimal(10,2)
);
